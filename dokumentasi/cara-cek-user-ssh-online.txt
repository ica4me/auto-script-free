journalctl --since "2 hours ago" -o cat | awk '
/Password auth succeeded for/ {
  line=$0
  pid=line
  sub(/^.*\[/,"",pid)
  sub(/\].*$/,"",pid)
  n=split(line,a,"'\''")
  user=a[2]
  online[pid]=user
}
/Exit \(/ {
  line=$0
  pid=line
  sub(/^.*\[/,"",pid)
  sub(/\].*$/,"",pid)
  delete online[pid]
}
END {
  found=0
  for (pid in online) {
    if (online[pid]=="lala") {
      print "lala ONLINE | PID:", pid
      found=1
    }
  }
  if (!found) print "lala OFFLINE"
}'
